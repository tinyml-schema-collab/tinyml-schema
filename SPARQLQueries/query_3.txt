SELECT ?rule ?input ?output 
    WHERE {
            ?rule a cep:Rule ;
                cep:head ?head ;
                cep:body ?body .
            ?head cep:operand ?output .
            ?body cep:eventOperation ?op .
            OPTIONAL { ?op cep:leftOperand ?input }
            OPTIONAL { ?op cep:rightOperand ?input }
            ?input a :ConveyorBeltState .
    }
